1 /** 
  2 * @class
  3 * @type singleton
  4 * @author zhangwen.cao[zhangwen.cao@aliyun-inc.com]
  5 * @description URL参数字串格式数据转换工具。
  6 */
  7 Alipw.utils.QueryString = function(){
  8 	var pub = 
  9 	/**
 10 	 * @lends Alipw.utils.QueryString
 11 	 */ 
 12 	{
 13 		encode:function(queryString,stringDecoder){
 14 			if(!Alipw.isSet(stringDecoder))stringDecoder = window.decodeURIComponent;
 15 			
 16 			var object = new Object();
 17 			var arr = queryString.split('&');
 18 			var data;
 19 			for(var i=0,len=arr.length;i<len;i++){
 20 				data = arr[i].split('=');
 21 				object[stringDecoder(data[0])] = stringDecoder(data[1]);
 22 			}
 23 			
 24 			return object;
 25 		},
 26 		decode:function(queryObject,stringEncoder){
 27 			if(!Alipw.isSet(stringEncoder))stringEncoder = window.encodeURIComponent;
 28 			
 29 			var array = new Array();
 30 			for(var i in queryObject){
 31 				array.push(stringEncoder(i) + '=' + stringEncoder(queryObject[i]));
 32 			}
 33 			
 34 			return array.join('&');
 35 		}
 36 	};
 37 	
 38 	return pub;
 39 }();
